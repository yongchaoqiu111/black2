# black2/models.py
from dataclasses import dataclass, field
from typing import Optional, List, Literal
from datetime import datetime
from enum import Enum

class RiskLevel(str, Enum):
    LOW = "low"          # 可交易，无需额外质押
    MEDIUM = "medium"    # 可交易，需提高质押比例
    HIGH = "high"        # 风险较高，建议拒绝或极高质押
    CRITICAL = "critical"  # 拒绝交易

class DisputeStatus(str, Enum):
    PENDING = "pending"
    RESOLVED = "resolved"
    REJECTED = "rejected"

@dataclass
class TransactionRecord:
    """单笔交易记录（链下存储，不上 Git）"""
    tx_id: str                     # 交易哈希
    counterparty: str              # 对方身份（Git 仓库地址 / eth 地址）
    amount: float                  # 交易金额
    status: Literal["success", "failed", "disputed"]
    timestamp: datetime
    evidence_hash: Optional[str] = None   # 证据 IPFS 哈希

@dataclass
class ArbitrationCase:
    """仲裁案例（存入公共仲裁库）"""
    case_id: str
    initiator: str                 # 发起方身份
    respondent: str                # 被诉方身份
    evidence_hash: str             # IPFS 证据哈希
    verdict: str                   # 裁决书摘要
    winner: Optional[str] = None   # 胜诉方
    timestamp: datetime = field(default_factory=datetime.utcnow)

@dataclass
class ReputationData:
    """
    b2p-repo.json 的数据结构（存储在 Git 仓库中）
    宏观脱敏数据，公开可读
    """
    # 身份标识
    agent_id: str                  # 唯一标识（公钥/ETH地址/Git仓库路径）
    
    # 宏观信誉指标（公开）
    total_score: float = 70.0      # 综合信誉分 0-100
    win_rate: float = 0.0          # 胜诉率（胜诉数/总仲裁数）
    total_transactions: int = 0    # 总交易次数
    total_volume: float = 0.0      # 总交易金额
    dispute_count: int = 0         # 被仲裁次数
    success_count: int = 0         # 交易成功次数
    
    # 摩擦系数（实时计算，非存储）
    friction_coefficient: float = 1.0
    
    # 时间衰减相关
    last_update: datetime = field(default_factory=datetime.utcnow)
    
    # 历史凭证（加密/哈希指针）
    arbitration_history: List[str] = field(default_factory=list)  # 仲裁案例 ID 列表
    recent_transactions_hash: Optional[str] = None  # 近期交易的 Merkle 根
    
    # 质押信息（与 X402 联动）
    staked_amount: float = 0.0     # 质押资金量
    staked_asset: str = "USDC"     # 质押资产类型

# black2/reputation.py
import math
from datetime import datetime, timedelta
from typing import Optional
from .models import ReputationData, RiskLevel

class ReputationEngine:
    """
    信誉计算引擎
    公式 v2.0：score = base_score * decay_factor + bonus
    """
    
    # 配置参数
    HALF_LIFE_DAYS = 30           # 半衰期（天）
    SCORE_DECAY_RATE = 0.05       # 每30天衰减5%
    
    # 阈值
    LOW_RISK_THRESHOLD = 70       # >=70 低风险
    MEDIUM_RISK_THRESHOLD = 50    # 50-69 中风险
    HIGH_RISK_THRESHOLD = 30      # 30-49 高风险
    # <30 严重风险
    
    @classmethod
    def calculate_time_decay(cls, last_update: datetime) -> float:
        """
        计算时间衰减因子
        衰减公式：decay = e^(-λ * days_since_update)
        """
        days_since = (datetime.utcnow() - last_update).days
        if days_since <= 0:
            return 1.0
        
        # λ = ln(2) / half_life
        lambda_val = math.log(2) / cls.HALF_LIFE_DAYS
        decay = math.exp(-lambda_val * days_since)
        return max(decay, 0.3)  # 最低不低于0.3
    
    @classmethod
    def calculate_friction_coefficient(cls, data: ReputationData) -> float:
        """
        计算摩擦系数
        系数越高，交易成本/质押要求越高
        """
        base = 1.0
        
        # 信誉分影响
        if data.total_score < 30:
            base += 1.5
        elif data.total_score < 50:
            base += 0.8
        elif data.total_score < 70:
            base += 0.3
        else:
            base -= 0.2   # 高分奖励
        
        # 胜诉率影响
        if data.win_rate < 30:
            base += 0.5
        elif data.win_rate > 80:
            base -= 0.2
        
        # 争议次数影响
        if data.dispute_count > 5:
            base += min(data.dispute_count * 0.05, 0.5)
        
        return max(base, 0.5)  # 不低于0.5
    
    @classmethod
    def get_risk_level(cls, data: ReputationData) -> RiskLevel:
        """根据信誉分判定风险等级"""
        score = data.total_score
        if score >= cls.LOW_RISK_THRESHOLD:
            return RiskLevel.LOW
        elif score >= cls.MEDIUM_RISK_THRESHOLD:
            return RiskLevel.MEDIUM
        elif score >= cls.HIGH_RISK_THRESHOLD:
            return RiskLevel.HIGH
        else:
            return RiskLevel.CRITICAL
    
    @classmethod
    def update_reputation(cls, data: ReputationData, 
                          success: bool, 
                          amount: float,
                          was_disputed: bool = False,
                          dispute_won: bool = False) -> ReputationData:
        """
        根据交易结果更新信誉数据
        """
        # 1. 应用时间衰减
        decay = cls.calculate_time_decay(data.last_update)
        data.total_score *= decay
        
        # 2. 更新交易统计
        data.total_transactions += 1
        data.total_volume += amount
        
        if success:
            data.success_count += 1
        else:
            # 交易失败，扣分
            data.total_score -= 5
            if was_disputed:
                data.dispute_count += 1
                data.total_score -= 10
                if dispute_won:
                    data.total_score += 8   # 胜诉返还部分
                    # 更新胜诉率
                    total_cases = len(data.arbitration_history)
                    if total_cases > 0:
                        wins = data.win_rate * total_cases / 100 + 1
                        data.win_rate = (wins / (total_cases + 1)) * 100
        
        # 3. 重新计算综合信誉分（加权）
        success_rate = data.success_count / max(data.total_transactions, 1)
        score_from_success = success_rate * 40  # 成功率贡献40分
        
        # 胜诉率贡献30分
        score_from_win = (data.win_rate / 100) * 30
        
        base_score = 30 + score_from_success + score_from_win
        
        # 争议惩罚
        dispute_penalty = min(data.dispute_count * 3, 20)
        final_score = max(base_score - dispute_penalty, 0)
        
        data.total_score = round(final_score, 2)
        data.friction_coefficient = cls.calculate_friction_coefficient(data)
        data.last_update = datetime.utcnow()
        
        return data

# black2/storage.py
import json
import os
from pathlib import Path
from typing import Optional
from .models import ReputationData

class StorageAdapter:
    """
    存储适配器 - 抽象 Git/IPFS 操作
    初版先用本地文件模拟
    """
    
    def __init__(self, workspace: str = "./b2p_data"):
        self.workspace = Path(workspace)
        self.workspace.mkdir(exist_ok=True)
    
    def pull_reputation(self, agent_id: str) -> Optional[ReputationData]:
        """从存储中拉取对方的信誉数据"""
        repo_path = self.workspace / f"{agent_id}.json"
        if not repo_path.exists():
            return None
        
        with open(repo_path, 'r') as f:
            data = json.load(f)
        return ReputationData(**data)
    
    def push_reputation(self, agent_id: str, data: ReputationData):
        """将信誉数据推送到存储"""
        repo_path = self.workspace / f"{agent_id}.json"
        with open(repo_path, 'w') as f:
            json.dump(data.__dict__, f, indent=2, default=str)
