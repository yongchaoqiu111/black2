2.2 碎片化核心 src/shard.py
python
import json
import random
import hashlib
import base64
from typing import List, Dict, Any
from config import Config

class ShardAlgorithm:
    """碎片化算法 - 你可以天天换这个类"""
    
    def __init__(self, version: int, private_seed: str):
        self.version = version
        self.seed = private_seed + str(version)
    
    def shard(self, data: Dict[str, Any]) -> List[str]:
        """把数据打碎成 5 个碎片"""
        # 1. 序列化
        json_str = json.dumps(data, sort_keys=True)
        
        # 2. 转成字符列表
        chars = list(json_str)
        
        # 3. 用你的私有种子打乱顺序
        random.seed(self.seed)
        random.shuffle(chars)
        
        # 4. 分成 5 份
        total = len(chars)
        shard_size = total // Config.TOTAL_SHARDS
        shards = []
        
        for i in range(Config.TOTAL_SHARDS):
            start = i * shard_size
            end = start + shard_size if i < Config.TOTAL_SHARDS - 1 else total
            shard_content = ''.join(chars[start:end])
            
            # 5. 加混淆头尾（每片看起来都一样）
            obfuscated = self._obfuscate(shard_content, i)
            shards.append(obfuscated)
        
        return shards
    
    def reconstruct(self, fragments: List[str]) -> Dict[str, Any]:
        """从 5 个碎片还原数据"""
        # 1. 去混淆
        cleaned = []
        for i, fragment in enumerate(fragments):
            content = self._deobfuscate(fragment, i)
            cleaned.append(content)
        
        # 2. 合并
        combined = ''.join(cleaned)
        
        # 3. 反打乱（需要知道打乱前的顺序）
        # 简化版：用同样的种子生成索引映射
        # 生产环境需要保存打乱前的索引映射
        
        # 4. 反序列化
        return json.loads(combined)
    
    def _obfuscate(self, content: str, shard_index: int) -> str:
        """混淆单个碎片"""
        # 加随机前缀后缀
        prefix = hashlib.md5(f"{self.seed}_{shard_index}".encode()).hexdigest()[:8]
        suffix = hashlib.md5(f"{shard_index}_{self.seed}".encode()).hexdigest()[:8]
        encoded = base64.b64encode(content.encode()).decode()
        return f"v{self.version}_{prefix}_{encoded}_{suffix}"
    
    def _deobfuscate(self, obfuscated: str, shard_index: int) -> str:
        """去混淆"""
        parts = obfuscated.split('_')
        # 格式: v{version}_{prefix}_{encoded}_{suffix}
        if len(parts) >= 4:
            encoded = parts[2]
            return base64.b64decode(encoded).decode()
        return ""


class AlgorithmRegistry:
    """算法版本管理 - 你每天加新版本"""
    
    _algorithms = {}
    
    @classmethod
    def register(cls, version: int, algorithm_class):
        cls._algorithms[version] = algorithm_class
    
    @classmethod
    def get(cls, version: int, private_seed: str):
        algorithm_cls = cls._algorithms.get(version)
        if not algorithm_cls:
            raise ValueError(f"Algorithm version {version} not found")
        return algorithm_cls(version, private_seed)
